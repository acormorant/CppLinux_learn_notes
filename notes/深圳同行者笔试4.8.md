1. 字符串分割和组合
```cpp
#include <iostream>
#include <string>
#include <vector>
using namespace std;

class Solution {
public:
    void Split(const string& src, const string& separator, vector<string>& res){
    	string str = src;
    	string sub_str;
    	string::size_type start = 0, index;
    	index = str.find_first_of(separator,start);
    	do{
    		if (index != string::npos){    
    			sub_str = str.substr(start,index-start);
    			res.emplace_back(sub_str);
    			start =index+separator.size();
    			index = str.find(separator,start);
    			if (start == string::npos) break;
    		}
    	}while(index != string::npos);
    
	    sub_str = str.substr(start);
    	res.emplace_back(sub_str);
    }

    void Connect(string& src, const string& separator, vector<string>& res){
        for(int i=0;i<res.size();i++){
            src.append(res[i]);
            src.append(separator);
        }
        src.erase(src.size()-separator.size()-1, separator.size());
    }

};
int main(){
    Solution temp;
    string str = "sajdasdaid@@sdasjdasd@@sadajsodasiii@@s@@ss@sad@@";
    vector<string> res;
    temp.Split(str, "@@", res);
    for(int i=0;i<res.size();i++){
        cout<<res[i]<<endl;
    }
    string resstr;
    temp.Connect(resstr, "&&", res);
    cout << resstr;
}
```

2. 圆桌问题
```cpp
#include<iostream>
#include<vector>
using namespace std;
class Solution {
public:
    int func(int n, int m) {
        vector<int> nums;
        for(int i=0;i<n;i++)
            nums.emplace_back(i);
        int index =0;
        while(nums.size() > 1){
            index=(index+m-1)%nums.size();
            nums.erase(nums.begin()+index);
        }
        return nums[0]+1;
    }
};
int main(){
    Solution temp;
    cout <<temp.func(10, 12);
}
```

4. 找重复数
```cpp
class Solution {
public:
    int findRepeatnum1(vector<int>& nums) {
        for(int i=0;i<nums.size();){
            if(nums[i] == i){
                i++;
                continue;
            }
            if(nums[nums[i]] == nums[i])return nums[i];
            swap(nums[i], nums[nums[i]]);
        }
        return -1;
    }
    int findRepeatnum2(vector<int>& nums) {
        set<int> repeatnums;
        for(auto & v: nums){
            if(repeatnums.find(v) != repeatnums.end())return v;
            repeatnums.emplace(v);
        }
        return 0;
    }
};
```
