# 父子进程 ls | wc -l
```cpp
#include<iosream>
#include<sys/types.h>
#include<unistd.h>
using namespace std;

int main(){
    int ret, fd[2];
    ret = pipe(fd);
    pid_t pid = fork();

    if(pid == -1){
        cerr << "fork failure\n";
        return -1;
    }
    else if(pid > 0){
        close(fd[0]);
        dup2(fd[1], STDOUT_FILENO);
        execlp("ls","ls", NULL);
        cerr << "ls failure\n";
    }
    else if(pid == 0){
        close(fd[1]);
        dup2(fd[0], STDIN_FILENO);
        execlp("wc", "wc", "-l", NULL);
        cerr << "wc -l failure\n";
    }
    return 0;
}
// 此程序固定父进程先执行完毕,故产生孤儿进程;
// 若想避免此情况可让父进程执行wc 子进程执行ls
```

# 兄弟进程 ls | wc -l
> 要求: 使用循环创建N个子进程模型创建兄弟进程，使用循环因子i标识，注意管道读写行为
> 注意!!!: 应该close无关进程的管道两端; 
```cpp
#include<iostream>
#include<sys/types.h>
#include<unistd.h>
#include<sys/wait.h>
using namespace std;

int main(){
    cout << "the number of child processes to be created is two\n";
    int number = 2, i, fd[2];

    pid_t pid,ret,wpid;
    ret = pipe(fd);
    if(ret == -1){
        cerr << "creat pipe failure\n";
        return -1;
    }
    for(i=0;i<number;i++){
        pid = fork();
        if(pid == 0){
            cout << i+1 << " child process is created\n";
            if(i == 0){
                sleep(number);
                close(fd[0]);
                dup2(fd[1], STDOUT_FILENO);
                execlp("ls","ls", NULL);
                cerr << "ls failure\n";                
            }
            else if(i == 1){
                sleep(number);
                close(fd[1]);
                dup2(fd[0], STDIN_FILENO);
                execlp("wc", "wc", "-l", NULL);
                cerr << "wc -l failure\n";
            }
            break;
        }
    }
    if(i == number){
        close(fd[0]);
        close(fd[1]);
        while((wpid = waitpid(-1,NULL,WNOHANG)) != -1){
            if(wpid>0)cout << "release successful\n";
            else if(wpid == 0 ){
                sleep(1);
                continue;
            }
        }
        cout << "parent process exit\n";
    }

    return 0;
}
```
