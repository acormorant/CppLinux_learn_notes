# fork多个子进程
```cpp
#include<iostream>
#include<sys/types.h>
#include<unistd.h>
using namespace std;

int main(){
    cout << "please enter the number of child processes to create: ";
    int number, i;
    cin >> number;

    pid_t pid;
    for(i=0;i<number;i++){
        pid = fork();
        if(pid == 0){
            sleep(i);
            cout << i+1 << " child process is created\n";
            break;
        }
    }
    if(i == number){
        sleep(number);
        cout << "parent process exit\n";
    }

    return 0;
}
```

# waitpid回收指定子进程
```cpp
#include<iostream>
#include<sys/types.h>
#include<unistd.h>
#include<sys/wait.h>
using namespace std;

int main(){
    cout << "please enter the number of child processes to create: ";
    int number, i, waitNum;
    cin >> number;
    cout << "please enter the number of child process to release: ";
    cin >> waitNum;
    if(waitNum<0 || waitNum>number)cout << "The entered number is out of bounds\n";

    pid_t pid, wpid;
    for(i=0;i<number;i++){
        pid = fork();
        if(pid == 0){
            sleep(i);
            cout << i+1 << " child process is created\n";
            break;
        }
        if(i == waitNum-1)wpid = pid;
    }
    if(i == number){
        sleep(number);
        wpid = waitpid(wpid, NULL, WNOHANG);
        if(wpid == -1)cout << waitNum <<" process release failure!\n";
        else if(wpid > 0 )cout << waitNum << " process release successful!\n";
        cout << "parent process exit\n";
    }

    return 0;
}
```

# waitpid 循环回收子进程
```cpp
#include<iostream>
#include<sys/types.h>
#include<unistd.h>
#include<sys/wait.h>
using namespace std;

int main(){
    cout << "please enter the number of child processes to create: ";
    int number, i;
    cin >> number;

    pid_t pid,wpid;
    for(i=0;i<number;i++){
        pid = fork();
        if(pid == 0){
            sleep(i);
            cout << i+1 << " child process is created\n";
            break;
        }
    }
    if(i == number){
        sleep(number);
        //不阻塞释放子进程
        while((wpid = waitpid(-1, NULL, WNOHANG)) != -1){
            if(wpid > 0)cout << "release failure\n";
            else if(wpid == 0 ){
                sleep(1);
                continue;
            }
        }
        cout << "parent process exit\n";
    }

    return 0;
}
```
